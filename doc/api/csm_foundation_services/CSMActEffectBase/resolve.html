Resolves the current CSMActEffect.

onSuccess - When the ActEffect resulves successfuly. onFailure - When the ActEffect receives a failed StatusCode. onException - When the ActEffect got interrupted by an exception. log - Wheter the exception should be logged.

@override
void resolve(
  Function(JObject success) onSuccess,
  Function(JObject failure, int statusCode) onFailure,
  Function(Object exception, StackTrace trace) onException,
) {
  if (_success != null) {
    onSuccess(_success);
    return;
  }
  if (_error != null) {
    onFailure(_error, _status ?? 0);
    return;
  }

  Object excep = _exception ?? Exception("No success, no error, no exception");
  StackTrace trace = _trace ?? StackTrace.current;
  onException(excep, trace);
}
